the constant density contour that contains 50% of the probability

library(car)
sigma<- matrix(c(2,sqrt(0.5),sqrt(0.5),1),nrow=2,ncol=2)
ellipse(center=c(0,2), shape=sigma, radius=sqrt(qchisq(0.5, df=2)),add=FALSE)
text(1,2,label="50%")
